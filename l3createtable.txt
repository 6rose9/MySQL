=> create table

Syntax
------

CREATE TABLE tablename{
    column1 datatype,
    column2 datatype,
    column3 datatype,
};

CREATE TABLE IF NOT EXISTS tablename{
    column1 datatype,
    column2 datatype,
    column3 datatype,
};

CREATE TABLE persons{
    id INT,
    firstname VARCHAR(size), //255
    lastname VARCHAR(100),
    city VARCHAR(255)
    address VARCHAR(255)
}
